        <<Abstract>>
        -renderer: ThemeRenderer
        +constructor(renderer)
        +render()*

        +click()
        +render()

        +showContent()
        +render()
    }

        <<Interface>>
        +drawButton()*
        +drawCard()*
    }

        +drawButton()
        +drawCard()
    }
    class ClassicRenderer {
        +drawButton()
        +drawCard()
    }

    ThemeRenderer <|.. ModernRenderer
    ThemeRenderer <|.. ClassicRenderer

    %% The Bridge Connection

    %% --- The Factory (Creator) ---

        <<Interface>>
        +createRenderer() ThemeRenderer

        +createRenderer()
    }
    class ClassicFactory {
        +createRenderer()
    }
