```ts
// =========================================================
// 🧪 ULTRA TEST SUITE: Design Pattern Validation
// =========================================================

console.log("\n🛑 STARTING COMPREHENSIVE PATTERN TESTS...\n");

// ---------------------------------------------------------
// TEST CASE 1: Director Polymorphism & Open/Closed Principle
// พิสูจน์ว่า: Director ไม่ผูกติดกับ Builder ตัวใดตัวหนึ่ง และทำงานได้หลายโหมด
// ---------------------------------------------------------
console.log("🧪 TEST 1: Director Polymorphism");

const objBuilder = new SDLCBuilder();
const mdBuilder = new MarkdownReportSDLCBuilder();

// ใช้ Director ตัวเดียว (Generic) สลับ Builder ไปมา
const director = new SDLCDirector(objBuilder);
const reportDirector = new SDLCDirector(mdBuilder);

console.log("   Step 1.1: Director drives SDLCBuilder (Object)");
director.constructStrictWaterfall(new Date());
const waterfallObj = objBuilder.build();
console.log(`      ✅ Result Type: ${waterfallObj.constructor.name}`); // ต้องได้ SDLCProjectDocument

// ใช้ Director ที่สอง สำหรับ Markdown Builder
console.log("   Step 1.2: Director drives MarkdownBuilder (Report)");

reportDirector.constructStrictWaterfall(new Date());
const waterfallReport = mdBuilder.build();
console.log(`      ✅ Result Type: ${waterfallReport.constructor.name}`); // ต้องได้ GeneratedReport

if (waterfallObj.techStack.length === waterfallReport['_techStack'].length) {
    console.log("      ✅ Logic Verification: Both products contain same data logic.");
} else {
    console.error("      ❌ Logic Error: Data mismatch between builders.");
}


// ---------------------------------------------------------
// TEST CASE 2: Builder State Reset (Isolation)
// พิสูจน์ว่า: เมื่อ build() เสร็จแล้ว ข้อมูลเก่าต้องหายไป ไม่ปนกับงานใหม่
// ---------------------------------------------------------
console.log("\n🧪 TEST 2: Builder State Isolation");

// งานที่ 1: Agile
reportDirector.constructAgileSprints(2, new Date());
const report1 = mdBuilder.build();

// งานที่ 2: MVP (ใช้ Builder ตัวเดิม!)
reportDirector.constructStartupMVP("TestApp");
const report2 = mdBuilder.build();

// ตรวจสอบ
if (report2.markdownContent.includes("Sprint 1")) {
    console.error("      ❌ FAILED: Data Leaked! Found Agile data in MVP report.");
} else {
    console.log("      ✅ PASSED: Builder reset correctly. No data leak.");
}


// ---------------------------------------------------------
// TEST CASE 3: Prototype Deep Copy Safety (The Critical Test)
// พิสูจน์ว่า: แก้ไข Clone แล้ว Master ไม่เปลี่ยน (Deep Copy)
// ---------------------------------------------------------
console.log("\n🧪 TEST 3: Prototype Deep Copy Safety");

// 3.1 สร้าง Master
const masterProject = waterfallObj; // เอาจาก Test 1
const originalReqCount = masterProject.functionalRequirements.length;
const firstReqId = masterProject.functionalRequirements[0].id;

// 3.2 Clone
console.log("   Step 3.1: Cloning Master...");
const forkedProject = masterProject.clone();

// 3.3 Mutate the Clone (แก้ไขตัวลูกแบบรุนแรง)
console.log("   Step 3.2: Mutating Clone (Adding reqs, changing array)...");
forkedProject.name = "HACKED PROJECT";
forkedProject.addSingleRequirement(new Requirement("HACK-001", "Malicious Code", "Critical"));
forkedProject.functionalRequirements[0].priority = "Low"; // แอบแก้ Priority ของตัวแรก

// 3.4 Verify Master (ตรวจสอบตัวแม่)
console.log("   Step 3.3: Verifying Master Integrity...");

const isNameSafe = masterProject.name !== "HACKED PROJECT";
const isCountSafe = masterProject.functionalRequirements.length === originalReqCount;
const isDeepObjectSafe = masterProject.functionalRequirements[0].priority !== "Low"; // ถ้า Deep Copy จริง ต้องเป็นค่าเดิม

if (isNameSafe && isCountSafe && isDeepObjectSafe) {
    console.log("      ✅ PASSED: Master is safe. Deep copy works perfectly.");
    console.log(`         - Master Priority: ${masterProject.functionalRequirements[0].priority}`);
    console.log(`         - Clone Priority:  ${forkedProject.functionalRequirements[0].priority}`);
} else {
    console.error("      ❌ FAILED: Reference Leak detected!");
    if (!isDeepObjectSafe) console.error("         - Object Reference was shared (Shallow Copy error).");
}


// ---------------------------------------------------------
// TEST CASE 4: Rich Product Behavior
// พิสูจน์ว่า: GeneratedReport เป็น Object ที่ฉลาด (Smart Product)
// ---------------------------------------------------------
console.log("\n🧪 TEST 4: Rich Product Behavior");

// เรียกใช้ Method ของ Product
console.log("   Step 4.1: Testing internal methods...");
try {
    // ลอง save (จำลอง)
    waterfallReport.saveToDisk("/var/www/docs");

    // ลอง print summary
    waterfallReport.printSummary();

    console.log("      ✅ PASSED: Product behaviors executed successfully.");
} catch (e) {
    console.error("      ❌ FAILED: Product method error.", e);
}

console.log("\n🏁 ALL TESTS COMPLETED.");
```
